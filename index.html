📜 Apps Script Code — Copy & Paste into Google Apps Script
Paste this into Extensions → Apps Script in your LeadFunnel sheet. Then deploy as a web app.
// ZagaPrime LeadFunnel — Google Apps Script Web App // Paste into Extensions > Apps Script, then Deploy > New Deployment > Web App // Execute as: Me | Who has access: Anyone const SHEET_ID = '1Sb4_mejZZR7UuOwzc3a6CK5gjfF5NeX__g6zN0G_boM'; const LEADS_SHEET = 'Sheet1'; const ACTIVITY_SHEET = 'Activity'; function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.openById(SHEET_ID); if (data.type === 'sync_leads') { syncLeads(ss, data.leads); } if (data.type === 'sync_activity') { syncActivity(ss, data.activity, data.leads); } if (data.type === 'full_sync') { syncLeads(ss, data.leads); syncActivity(ss, data.activity, data.leads); formatSheets(ss); } return ContentService.createTextOutput( JSON.stringify({success: true, timestamp: new Date().toISOString()}) ).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput( JSON.stringify({success: false, error: err.message}) ).setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService.createTextOutput( JSON.stringify({status: 'ZagaPrime LeadFunnel webhook is live', time: new Date().toISOString()}) ).setMimeType(ContentService.MimeType.JSON); } function syncLeads(ss, leads) { let sheet = ss.getSheetByName(LEADS_SHEET); if (!sheet) sheet = ss.insertSheet(LEADS_SHEET); const headers = ['priority','business_name','owner_name','category','phone', 'address','city','county','state','zip','website_status', 'online_presence','pitch_angle','tier','years_in_business','notes', 'stage_research','stage_first_call','stage_email_sent','stage_discovery_call', 'stage_proposal_sent','stage_follow_up','stage_decision','stage_notes', 'deal_value','last_contacted','next_action','next_action_date','assigned_to','date_added']; sheet.clearContents(); sheet.getRange(1, 1, 1, headers.length).setValues([headers]); const rows = leads.map(l => headers.map(h => { if (h === 'stage_research') return l._stages?.research?.value || ''; if (h === 'stage_first_call') return l._stages?.first_call?.value || ''; if (h === 'stage_email_sent') return l._stages?.email_sent?.value || ''; if (h === 'stage_discovery_call') return l._stages?.discovery_call?.value || ''; if (h === 'stage_proposal_sent') return l._stages?.proposal_sent?.value || ''; if (h === 'stage_follow_up') return l._stages?.follow_up?.value || ''; if (h === 'stage_decision') return l._stages?.decision?.value || ''; if (h === 'stage_notes') return l._stages ? Object.values(l._stages).map(s=>s.note).filter(Boolean).join(' | ') : ''; return l[h] !== undefined ? String(l[h]) : ''; })); if (rows.length > 0) { sheet.getRange(2, 1, rows.length, headers.length).setValues(rows); } formatSheets(ss); } function syncActivity(ss, activity, leads) { let sheet = ss.getSheetByName(ACTIVITY_SHEET); if (!sheet) sheet = ss.insertSheet(ACTIVITY_SHEET); const headers = ['business_name','category','city','stage','outcome','note','timestamp']; sheet.clearContents(); sheet.getRange(1, 1, 1, headers.length).setValues([headers]); const rows = []; Object.entries(activity || {}).forEach(([leadId, act]) => { const lead = leads.find(l => l.id === leadId) || {}; Object.entries(act.stages || {}).forEach(([stageId, sv]) => { rows.push([lead.name||'', lead.category||'', lead.city||'', stageId, sv.value||'', sv.note||'', sv.timestamp||'']); }); }); if (rows.length > 0) { sheet.getRange(2, 1, rows.length, headers.length).setValues(rows); } } function formatSheets(ss) { const sheet = ss.getSheetByName(LEADS_SHEET); if (!sheet) return; const lastCol = sheet.getLastColumn(); if (lastCol < 1) return; // Header row: dark background, white text, bold const headerRange = sheet.getRange(1, 1, 1, lastCol); headerRange.setBackground('#1e293b'); headerRange.setFontColor('#ffffff'); headerRange.setFontWeight('bold'); headerRange.setFontSize(11); // Freeze header row sheet.setFrozenRows(1); // Color code tier column (col 14 = tier) const lastRow = sheet.getLastRow(); if (lastRow > 1) { for (let i = 2; i <= lastRow; i++) { const tierCell = sheet.getRange(i, 14); const tier = tierCell.getValue(); if (tier == 1) sheet.getRange(i, 1, 1, lastCol).setBackground('#fef2f2'); else if (tier == 2) sheet.getRange(i, 1, 1, lastCol).setBackground('#fffbeb'); else if (tier == 3) sheet.getRange(i, 1, 1, lastCol).setBackground('#eff6ff'); } } // Auto-resize all columns sheet.autoResizeColumns(1, lastCol); }
